Зная, что разным людям свойственно либо давать заниженные, либо завышенные оценки, нормализуем оценки фильмов по каждому индивиду. В качестве метода нормализации выберем минимаксную нормализацию x_i=(x_i-min(x))/(max(x) - min(x))*12. https://wiki.loginom.ru/articles/data-normalization.html

data$max <- apply(data[,1:101], 1, max, na.rm=TRUE)
data$min <- apply(data[,1:101], 1, min, na.rm=TRUE)

data[,1:101] <- apply(data[,1:101], 2, function(x) {(x - data$min) / (data$max - data$min) * 12})
